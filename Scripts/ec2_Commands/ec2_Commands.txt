sudo yum update
sudo yum install java-1.8.0  --> To install java
wget https://downloads.apache.org/kafka/3.9.2/kafka_2.12-3.9.2.tgz --> kafka link
tar -xzf kafka_2.12-3.9.2.tgz --> To unzip kafka file
sudo mv kafka_2.12-3.9.2 /usr/local/kafka --> to move unziped kafka file in to provided path
cd /usr/local/kafka
sudo nano config/server.properties --> To open the config file

bin/zookeeper-server-start.sh config/zookeeper.properties --> To start zookeeper
